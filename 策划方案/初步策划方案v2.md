```sql
-- ============================================================
-- 排产调度模块 (Scheduler Module)
-- ============================================================

-- 批次信息
CREATE TABLE Batches (
    batch_id        TEXT PRIMARY KEY,           -- 批次号
    part_no         TEXT NOT NULL,              -- 图号
    part_name       TEXT,                       -- 名称
    quantity        INTEGER NOT NULL,           -- 数量
    due_date        DATE,                       -- 交期 (手填)
    priority        TEXT DEFAULT 'normal',      -- 优先级: normal/urgent/critical
    ready_status    TEXT DEFAULT 'no',          -- 齐套状态: yes/no/partial
    status          TEXT DEFAULT 'pending',     -- 状态: pending/scheduled/processing/completed/cancelled
    remark          TEXT,                       -- 备注
    created_at      DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (part_no) REFERENCES Parts(part_no)
);

-- 批次工序 (从模板复制，可修改)
CREATE TABLE BatchOperations (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,  -- 自增主键
    op_code         TEXT NOT NULL UNIQUE,       -- 工序编码 (batch_id + seq 生成，如 B001_05)
    batch_id        TEXT NOT NULL,              -- 所属批次
    piece_id        TEXT,                       -- 零件流水号 (预留，用于单件追溯)
    seq             INTEGER NOT NULL,           -- 工序号
    op_type_id      TEXT,                       -- 工种ID
    op_type_name    TEXT NOT NULL,              -- 工种名称
    source          TEXT DEFAULT 'internal',    -- 归属: internal/external
    machine_id      TEXT,                       -- 设备编号 (内部工序)
    operator_id     TEXT,                       -- 操机人员 (内部工序)
    supplier_id     TEXT,                       -- 供应商ID (外部工序)
    setup_hours     REAL DEFAULT 0,             -- 换型时间
    unit_hours      REAL DEFAULT 0,             -- 单件工时
    ext_days        REAL,                       -- 外部周期(天)
    status          TEXT DEFAULT 'pending',     -- 状态: pending/scheduled/processing/completed/skipped
    created_at      DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (batch_id) REFERENCES Batches(batch_id) ON DELETE CASCADE,
    FOREIGN KEY (machine_id) REFERENCES Machines(machine_id),
    FOREIGN KEY (operator_id) REFERENCES Operators(operator_id),
    FOREIGN KEY (supplier_id) REFERENCES Suppliers(supplier_id),
    UNIQUE (batch_id, seq, piece_id)            -- 复合唯一约束
);

-- 排程结果
CREATE TABLE Schedule (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    op_id           INTEGER NOT NULL,           -- 工序ID (BatchOperations.id)
    machine_id      TEXT,                       -- 设备 (内部工序)
    operator_id     TEXT,                       -- 操机人员 (内部工序)
    start_time      DATETIME NOT NULL,          -- 开始时间
    end_time        DATETIME NOT NULL,          -- 结束时间

## 四、工艺路线解析器

### 4.1 解析逻辑

```
┌─────────────────────────────────────────────────────────────┐
│                    工艺路线解析流程                          │
└─────────────────────────────────────────────────────────────┘

输入: "5数铣 10钳-20数车、25数铣，30钳 35标印40总检45表处理"
      (包含各种分隔符和空格)

Step 0: 预处理
─────────────────────────────────────────────────────────────
  1. 去除首尾空格
  2. 统一分隔符（空格、逗号、顿号、破折号 → 移除）
  3. 保留数字和汉字
  
  结果: "5数铣10钳20数车25数铣30钳35标印40总检45表处理"

Step 1: 正则匹配
─────────────────────────────────────────────────────────────
  模式: (\d+)([^\d]+)

  匹配结果:
  [(5, "数铣"), (10, "钳"), (20, "数车"), (25, "数铣"), 
   (30, "钳"), (35, "标印"), (40, "总检"), (45, "表处理")]

Step 2: 工种识别
─────────────────────────────────────────────────────────────
  查询 OpTypes 表，判断每个工种是否为本单元能力

  数铣 → category='internal' → internal
  钳   → category='internal' → internal
  数车 → category='internal' → internal
  标印 → category='external' → external
  总检 → category='external' → external
  表处理 → category='external' → external

  未识别工种 → 警告，默认标记为 external

Step 3: 识别连续外部工序组
─────────────────────────────────────────────────────────────
  遍历工序列表，将连续的外部工序归为一组

  [5数铣, 10钳, 20数车, 25数铣, 30钳] → 各自独立
  [35标印, 40总检, 45表处理] → 连续外部组 EXT_1

Step 4: 生成解析报告
─────────────────────────────────────────────────────────────
  {
    "success": true,
    "operations": [...],
    "external_groups": [...],
    "warnings": ["工种'表处理'未在系统中配置，已默认标记为外部工序"],
    "stats": {
      "total": 8,
      "internal": 5,
      "external": 3,
      "unknown": 1
    }
  }
```

### 4.2 解析器代码

```python
# core/services/process/route_parser.py

import re
from typing import List, Optional, Tuple, Dict, Any
from dataclasses import dataclass, field
from enum import Enum

class ParseStatus(Enum):
    SUCCESS = "success"
    PARTIAL = "partial"  # 有警告但可用
    FAILED = "failed"

@dataclass
class ParsedOperation:
    """解析后的工序"""
    seq: int                        # 工序号
    op_type_name: str              # 工种名称
    source: str                     # internal/external
    op_type_id: Optional[str] = None
    supplier_id: Optional[str] = None
    default_days: Optional[float] = None
    ext_group_id: Optional[str] = None
    is_recognized: bool = True      # 是否识别的工种

@dataclass
class ExternalGroup:
    """外部工序组"""
    group_id: str
    start_seq: int
    end_seq: int
    operations: List[ParsedOperation] = field(default_factory=list)

@dataclass
class ParseResult:
    """解析结果"""
    status: ParseStatus
    operations: List[ParsedOperation]
    external_groups: List[ExternalGroup]
    warnings: List[str]
    errors: List[str]
    stats: Dict[str, int]
    original_input: str
    normalized_input: str

class RouteParser:
    """工艺路线解析器"""
    
    # 需要移除的分隔符
    SEPARATORS = r'[\s,，、\-\—\→\->]+'
    
    def __init__(self, op_types_repo, suppliers_repo, logger=None):
        """
        Args:
            op_types_repo: 工种数据访问对象
            suppliers_repo: 供应商数据访问对象
            logger: 日志记录器
        """
        self.op_types_repo = op_types_repo
        self.suppliers_repo = suppliers_repo
        self.logger = logger
    
    def parse(self, route_string: str, part_no: str) -> ParseResult:
        """
        解析工艺路线字符串
        
        Args:
            route_string: 原始工艺路线字符串
            part_no: 图号 (用于生成组ID)
        
        Returns:
            ParseResult 解析结果
        """
        warnings = []
        errors = []
        original_input = route_string
        
        # Step 0: 预处理
        normalized = self._preprocess(route_string)
        
        if not normalized:
            return ParseResult(
                status=ParseStatus.FAILED,
                operations=[],
                external_groups=[],
                warnings=[],
                errors=["工艺路线为空或格式无效"],
                stats={"total": 0, "internal": 0, "external": 0, "unknown": 0},
                original_input=original_input,
                normalized_input=""
            )
        
        # 加载配置
        op_types = {ot.name: ot for ot in self.op_types_repo.get_all()}
        suppliers = self._build_supplier_map()
        
        # Step 1: 正则匹配
        pattern = r'(\d+)([^\d]+)'
        matches = re.findall(pattern, normalized)
        
        if not matches:
            return ParseResult(
                status=ParseStatus.FAILED,
                operations=[],
                external_groups=[],
                warnings=[],
                errors=["无法识别工艺路线格式，请使用'工序号+工种名'格式，如'5数铣10钳20数车'"],
                stats={"total": 0, "internal": 0, "external": 0, "unknown": 0},
                original_input=original_input,
                normalized_input=normalized
            )
        
        # Step 2: 工种识别
        operations = []
        stats = {"total": 0, "internal": 0, "external": 0, "unknown": 0}
        seen_seqs = set()
        
        for seq_str, op_type_name in matches:
            seq = int(seq_str)
            op_type_name = op_type_name.strip()
            
            # 检查工序号重复
            if seq in seen_seqs:
                warnings.append(f"工序号 {seq} 重复出现，将保留第一个")
                continue
            seen_seqs.add(seq)
            
            stats["total"] += 1
            
            op_type = op_types.get(op_type_name)
            
            if op_type:
                is_internal = op_type.category == 'internal'
                is_recognized = True
            else:
                # 未识别的工种，默认为外部
                is_internal = False
                is_recognized = False
                warnings.append(f"工种'{op_type_name}'未在系统中配置，已默认标记为外部工序")
                stats["unknown"] += 1
            
            source = 'internal' if is_internal else 'external'
            if is_internal:
                stats["internal"] += 1
            else:
                stats["external"] += 1
            
            op = ParsedOperation(
                seq=seq,
                op_type_name=op_type_name,
                op_type_id=op_type.op_type_id if op_type else None,
                source=source,
                is_recognized=is_recognized
            )
            
            # 外部工序：匹配供应商
            if not is_internal:
                supplier_info = suppliers.get(op_type_name)
                if supplier_info:
                    op.supplier_id = supplier_info[0]
                    op.default_days = supplier_info[1]
                else:
                    op.default_days = 1.0  # 默认1天
            
            operations.append(op)
        
        # 按工序号排序
        operations.sort(key=lambda x: x.seq)
        
        # Step 3: 识别连续外部工序组
        external_groups = self._identify_external_groups(operations, part_no)
        
        # 确定解析状态
        if errors:
            status = ParseStatus.FAILED
        elif warnings:
            status = ParseStatus.PARTIAL
        else:
            status = ParseStatus.SUCCESS
        
        result = ParseResult(
            status=status,
            operations=operations,
            external_groups=external_groups,
            warnings=warnings,
            errors=errors,
            stats=stats,
            original_input=original_input,
            normalized_input=normalized
        )
        
        # 记录日志
        if self.logger:
            self.logger.info(
                f"Route parsed for {part_no}: "
                f"{stats['total']} ops ({stats['internal']} internal, "
                f"{stats['external']} external, {stats['unknown']} unknown)"
            )
        
        return result
    
    def _preprocess(self, route_string: str) -> str:
        """预处理工艺路线字符串"""
        if not route_string:
            return ""
        
        # 1. 去除首尾空格
        result = route_string.strip()
        
        # 2. 移除常见分隔符
        result = re.sub(self.SEPARATORS, '', result)
        
        # 3. 全角数字转半角
        full_to_half = str.maketrans('０１２３４５６７８９', '0123456789')
        result = result.translate(full_to_half)
        
        return result
    
    def _build_supplier_map(self) -> dict:
        """构建 工种名 -> (supplier_id, default_days) 映射"""
        result = {}
        for supplier in self.suppliers_repo.get_all():
            if supplier.op_type_id:
                op_type = self.op_types_repo.get_by_id(supplier.op_type_id)
                if op_type:
                    result[op_type.name] = (supplier.supplier_id, supplier.default_days)
        return result
    
    def _identify_external_groups(self, operations: List[ParsedOperation], 
                                   part_no: str) -> List[ExternalGroup]:
        """识别连续外部工序组"""
        groups = []
        current_group = None
        group_counter = 1
        
        for op in operations:
            if op.source == 'external':
                if current_group is None:
                    # 开始新组
                    group_id = f"{part_no}_EXT_{group_counter}"
                    current_group = ExternalGroup(
                        group_id=group_id,
                        start_seq=op.seq,
                        end_seq=op.seq,
                        operations=[op]
                    )
                    op.ext_group_id = group_id
                else:
                    # 继续当前组
                    current_group.end_seq = op.seq
                    current_group.operations.append(op)
                    op.ext_group_id = current_group.group_id
            else:
                if current_group is not None:
                    # 结束当前组
                    groups.append(current_group)
                    current_group = None
                    group_counter += 1
        
        # 处理最后一组
        if current_group is not None:
            groups.append(current_group)
        
        return groups
    
    def validate_format(self, route_string: str) -> Tuple[bool, str]:
        """
        快速验证格式是否可解析（不做完整解析）
        
        Returns:
            (是否有效, 提示信息)
        """
        if not route_string or not route_string.strip():
            return False, "工艺路线不能为空"
        
        normalized = self._preprocess(route_string)
        pattern = r'(\d+)([^\d]+)'
        matches = re.findall(pattern, normalized)
        
        if not matches:
            return False, "格式无效，请使用'工序号+工种名'格式"
        
        if len(matches) < 1:
            return False, "至少需要一道工序"
        
        return True, f"格式有效，识别到 {len(matches)} 道工序"
```

## 五、外部工序删除规则

### 5.1 规则定义

```
┌─────────────────────────────────────────────────────────────┐
│                  外部工序删除规则                            │
└─────────────────────────────────────────────────────────────┘

规则1: 首部连续外部工序 → 可删除
─────────────────────────────────────────────────────────────
  原: [5外][10外][15内][20内]...
  删除 [5外][10外] → [15内][20内]...
  ✅ 允许，不影响内部工序连续性

规则2: 尾部连续外部工序 → 可删除
─────────────────────────────────────────────────────────────
  原: ...[20内][25内][30外][35外][40外]
  删除 [30外][35外][40外] → ...[20内][25内]
  ✅ 允许，不影响内部工序连续性

规则3: 中间外部工序 → 不可单独删除
─────────────────────────────────────────────────────────────
  原: [5内][10内][15外][20内][25内]
  删除 [15外] → 工序10和工序20之间断档
  ❌ 不允许

规则4: 中间连续外部工序 → 整组都不可删
─────────────────────────────────────────────────────────────
  原: [5内][10外][15外][20外][25内]
  删除任意外部 → 工序5和工序25之间断档
  ❌ 不允许

判定公式:
─────────────────────────────────────────────────────────────

  可删除条件：删除后，所有内部工序之间没有外部工序断点

  具体算法：
  1. 标记要删除的工序
  2. 遍历剩余工序
  3. 检查任意两个相邻内部工序之间是否有未删除的外部工序
  4. 如果有，则不允许删除
```

### 5.2 删除验证器代码

```python
# core/services/process/deletion_validator.py

from typing import List, Tuple, Set
from dataclasses import dataclass
from enum import Enum

class ValidationResult(Enum):
    ALLOWED = "allowed"
    DENIED = "denied"
    WARNING = "warning"

@dataclass
class Operation:
    seq: int
    source: str  # internal/external
    status: str = 'active'

@dataclass
class DeletionCheckResult:
    """删除检查结果"""
    result: ValidationResult
    can_delete: bool
    message: str
    affected_ops: List[int] = None  # 受影响的工序

class DeletionValidator:
    """外部工序删除验证器"""
    
    def can_delete(self, 
                   operations: List[Operation], 
                   to_delete: List[int]) -> DeletionCheckResult:
        """
        检查是否可以删除指定的外部工序
        
        Args:
            operations: 完整工序列表（按seq排序）
            to_delete: 要删除的工序seq列表
        
        Returns:
            DeletionCheckResult 检查结果
        """
        to_delete_set = set(to_delete)
        
        # 过滤掉已删除的工序
        active_ops = [op for op in operations if op.status == 'active']
        
        # 检查：要删除的必须都是外部工序
        for seq in to_delete:
            op = self._find_op(active_ops, seq)
            if op is None:
                return DeletionCheckResult(
                    result=ValidationResult.DENIED,
                    can_delete=False,
                    message=f"工序 {seq} 不存在或已删除"
                )
            if op.source == 'internal':
                return DeletionCheckResult(
                    result=ValidationResult.DENIED,
                    can_delete=False,
                    message=f"工序 {seq} 是内部工序，不能删除。只能删除外部工序。"
                )
        
        # 构建删除后的工序列表
        remaining = [op for op in active_ops if op.seq not in to_delete_set]
        
        # 如果删除后没有工序了
        if not remaining:
            return DeletionCheckResult(
                result=ValidationResult.WARNING,
                can_delete=True,
                message="删除后将没有任何工序，确定继续？"
            )
        
        # 如果没有剩余内部工序，允许删除
        internal_ops = [op for op in remaining if op.source == 'internal']
        if len(internal_ops) == 0:
            return DeletionCheckResult(
                result=ValidationResult.WARNING,
                can_delete=True,
                message="删除后将没有内部工序，确定继续？"
            )
        
        if len(internal_ops) == 1:
            return DeletionCheckResult(
                result=ValidationResult.ALLOWED,
                can_delete=True,
                message="可以删除"
            )
        
        # 检查内部工序之间是否有外部工序断档
        internal_ops.sort(key=lambda x: x.seq)
        
        for i in range(len(internal_ops) - 1):
            current = internal_ops[i]
            next_op = internal_ops[i + 1]
            
            # 检查两个内部工序之间是否有未删除的外部工序
            between_ops = [
                op for op in remaining 
                if current.seq < op.seq < next_op.seq and op.source == 'external'
            ]
            
            if between_ops:
                between_seqs = [op.seq for op in between_ops]
                return DeletionCheckResult(
                    result=ValidationResult.DENIED,
                    can_delete=False,
                    message=(
                        f"无法删除：工序 {current.seq} 和 {next_op.seq} 之间"
                        f"存在未删除的外部工序 {between_seqs}，"
                        f"删除后会导致工艺断档。"
                        f"如需删除，请一并删除这些工序。"
                    ),
                    affected_ops=between_seqs
                )
        
        return DeletionCheckResult(
            result=ValidationResult.ALLOWED,
            can_delete=True,
            message="可以删除"
        )
    
    def _find_op(self, operations: List[Operation], seq: int) -> Operation:
        """查找指定工序号的工序"""
        for op in operations:
            if op.seq == seq:
                return op
        return None
    
    def get_deletable_external_ops(self, operations: List[Operation]) -> List[int]:
        """
        获取所有可删除的外部工序
        
        Returns:
            可删除的工序seq列表
        """
        active_ops = [op for op in operations if op.status == 'active']
        external_ops = [op for op in active_ops if op.source == 'external']
        deletable = []
        
        for op in external_ops:
            result = self.can_delete(operations, [op.seq])
            if result.can_delete:
                deletable.append(op.seq)
        
        return deletable
    
    def get_deletion_groups(self, operations: List[Operation]) -> List[List[int]]:
        """
        获取可删除的外部工序组
        （首部连续外部工序组、尾部连续外部工序组）
        
        Returns:
            可删除的工序组列表，每组是一个seq列表
        """
        active_ops = sorted(
            [op for op in operations if op.status == 'active'],
            key=lambda x: x.seq
        )
        
        if not active_ops:
            return []
        
        groups = []
        
        # 检查首部连续外部工序
        head_group = []
        for op in active_ops:
            if op.source == 'external':
                head_group.append(op.seq)
            else:
                break
        if head_group:
            groups.append(head_group)
        
        # 检查尾部连续外部工序
        tail_group = []
        for op in reversed(active_ops):
            if op.source == 'external':
                tail_group.insert(0, op.seq)
            else:
                break
        if tail_group and tail_group != head_group:
            groups.append(tail_group)
        
        return groups
```

## 六、连续外部工序合并周期

### 6.1 处理逻辑

```
┌─────────────────────────────────────────────────────────────┐
│                连续外部工序合并周期                          │
└─────────────────────────────────────────────────────────────┘

### 7.1 多策略排序系统

```
┌─────────────────────────────────────────────────────────────┐
│                    排序策略系统                              │
└─────────────────────────────────────────────────────────────┘

策略1: 优先级优先 (priority_first)
─────────────────────────────────────────────────────────────
  排序规则:
    1. 优先级: critical > urgent > normal
    2. 同优先级按交期升序
    3. 同交期按批次号升序
  
  适用场景: 有明确紧急件需要优先处理

策略2: 交期优先 (due_date_first)
─────────────────────────────────────────────────────────────
  排序规则:
    1. 交期升序（越早越优先）
    2. 无交期的排最后
    3. 同交期按优先级
    4. 同优先级按批次号
  
  适用场景: 按期交付压力大

策略3: 权重混合 (weighted)
─────────────────────────────────────────────────────────────
  计算公式:
    score = priority_weight × priority_score 
          + due_weight × due_score
          + ready_weight × ready_score
  
  默认权重:
    priority_weight = 0.4
    due_weight = 0.5
    ready_weight = 0.1
  
  分数计算:
    priority_score: critical=100, urgent=60, normal=20
    due_score: 100 - min(days_left, 100)  # 越紧迫分越高
    ready_score: ready=100, partial=50, not_ready=0
  
  适用场景: 综合考虑多因素

策略4: 先进先出 (fifo)
─────────────────────────────────────────────────────────────
  排序规则:
    1. 按批次创建时间升序
    2. 同时间按批次号升序
  
  适用场景: 简单公平调度


策略配置表:
─────────────────────────────────────────────────────────────

│ due_weight           │ 0.5            │ 权重模式-交期权重  │

### 7.2 策略实现代码

```python
# core/algorithms/sort_strategies.py

from abc import ABC, abstractmethod
from datetime import datetime, date
from typing import List, Dict, Any
from dataclasses import dataclass
from enum import Enum

class SortStrategy(Enum):
    PRIORITY_FIRST = "priority_first"
    DUE_DATE_FIRST = "due_date_first"
    WEIGHTED = "weighted"
    FIFO = "fifo"

@dataclass
class BatchForSort:
    """用于排序的批次信息"""
    batch_id: str
    priority: str  # normal/urgent/critical
    due_date: date = None
    ready_status: str = 'no'  # yes/no/partial
    created_at: datetime = None

class BaseSortStrategy(ABC):
    """排序策略基类"""
    
    @abstractmethod
    def sort(self, batches: List[BatchForSort]) -> List[BatchForSort]:
        """执行排序"""
        pass
    
    @abstractmethod
    def get_name(self) -> str:
        """策略名称"""
        pass

class PriorityFirstStrategy(BaseSortStrategy):
    """优先级优先策略"""
    
    PRIORITY_ORDER = {'critical': 0, 'urgent': 1, 'normal': 2}
    
    def sort(self, batches: List[BatchForSort]) -> List[BatchForSort]:
        def sort_key(batch: BatchForSort):
            priority_rank = self.PRIORITY_ORDER.get(batch.priority, 99)
            due_rank = batch.due_date if batch.due_date else date.max
            return (priority_rank, due_rank, batch.batch_id)
        
        return sorted(batches, key=sort_key)
    
    def get_name(self) -> str:
        return "优先级优先"

class DueDateFirstStrategy(BaseSortStrategy):
    """交期优先策略"""
    
    PRIORITY_ORDER = {'critical': 0, 'urgent': 1, 'normal': 2}
    
    def sort(self, batches: List[BatchForSort]) -> List[BatchForSort]:
        def sort_key(batch: BatchForSort):
            # 无交期的排最后
            has_due = 0 if batch.due_date else 1
            due_rank = batch.due_date if batch.due_date else date.max
            priority_rank = self.PRIORITY_ORDER.get(batch.priority, 99)
            return (has_due, due_rank, priority_rank, batch.batch_id)
        
        return sorted(batches, key=sort_key)
    
    def get_name(self) -> str:
        return "交期优先"

class WeightedStrategy(BaseSortStrategy):
    """权重混合策略"""
    
    PRIORITY_SCORE = {'critical': 100, 'urgent': 60, 'normal': 20}
    READY_SCORE = {'yes': 100, 'partial': 50, 'no': 0}
    
    def __init__(self, priority_weight=0.4, due_weight=0.5, ready_weight=0.1):
        self.priority_weight = priority_weight
        self.due_weight = due_weight
        self.ready_weight = ready_weight
    
    def sort(self, batches: List[BatchForSort]) -> List[BatchForSort]:
        today = date.today()
        
        def calc_score(batch: BatchForSort) -> float:
            # 优先级分数
            priority_score = self.PRIORITY_SCORE.get(batch.priority, 0)
            
            # 交期分数 (越紧迫分越高)
            if batch.due_date:
                days_left = (batch.due_date - today).days
                due_score = 100 - min(max(days_left, 0), 100)
            else:
                due_score = 0  # 无交期不紧迫
            
            # 齐套分数
            ready_score = self.READY_SCORE.get(batch.ready_status, 0)
            
            # 加权总分
            total = (
                self.priority_weight * priority_score +
                self.due_weight * due_score +
                self.ready_weight * ready_score
            )
            
            return total
        
        # 按分数降序排序
        return sorted(batches, key=lambda b: (-calc_score(b), b.batch_id))
    
    def get_name(self) -> str:
        return "权重混合"

class FIFOStrategy(BaseSortStrategy):
    """先进先出策略"""
    
    def sort(self, batches: List[BatchForSort]) -> List[BatchForSort]:
        def sort_key(batch: BatchForSort):
            created = batch.created_at if batch.created_at else datetime.max
            return (created, batch.batch_id)
        
        return sorted(batches, key=sort_key)
    
    def get_name(self) -> str:
        return "先进先出"

class StrategyFactory:
    """策略工厂"""
    
    _strategies = {
        SortStrategy.PRIORITY_FIRST: PriorityFirstStrategy,
        SortStrategy.DUE_DATE_FIRST: DueDateFirstStrategy,
        SortStrategy.WEIGHTED: WeightedStrategy,
        SortStrategy.FIFO: FIFOStrategy,
    }
    
    @classmethod
    def create(cls, strategy: SortStrategy, **kwargs) -> BaseSortStrategy:
        """创建策略实例"""
        strategy_class = cls._strategies.get(strategy)
        if not strategy_class:
            raise ValueError(f"Unknown strategy: {strategy}")
        
        if strategy == SortStrategy.WEIGHTED:
            return strategy_class(**kwargs)
        return strategy_class()
    
    @classmethod
    def get_available_strategies(cls) -> List[Dict[str, str]]:
        """获取所有可用策略"""
        return [
            {"key": s.value, "name": cls.create(s).get_name()}
            for s in SortStrategy
        ]
```

### 7.3 双重资源约束算法

```python
# core/algorithms/greedy.py

from datetime import datetime, timedelta
from typing import Dict, List, Optional
from dataclasses import dataclass
import logging

from .sort_strategies import (
    SortStrategy, StrategyFactory, BatchForSort
)

@dataclass
class ScheduleResult:
    """排程结果"""
    op_id: int                      # BatchOperations.id
    op_code: str                    # 工序编码
    batch_id: str
    seq: int
    machine_id: Optional[str] = None
    operator_id: Optional[str] = None
    start_time: Optional[datetime] = None
    end_time: Optional[datetime] = None
    source: str = 'internal'

@dataclass
class ScheduleSummary:
    """排程摘要"""
    success: bool
    total_ops: int
    scheduled_ops: int
    failed_ops: int
    warnings: List[str]
    errors: List[str]
    duration_seconds: float

class GreedyScheduler:
    """贪心排产算法"""
    
    def __init__(self, calendar_service, config_service, logger=None):
        self.calendar = calendar_service
        self.config = config_service
        self.logger = logger or logging.getLogger(__name__)
    
    def schedule(self, 
                 operations: List, 
                 batches: Dict,
                 strategy: SortStrategy = None,
                 strategy_params: Dict = None) -> tuple:
        """
        执行排产
        
        Args:
            operations: 待排工序列表
            batches: 批次信息字典 {batch_id: Batch}
            strategy: 排序策略 (可选，默认从配置读取)
            strategy_params: 策略参数 (用于权重策略)
        
        Returns:
            (排程结果列表, 排程摘要)
        """
        start_time = datetime.now()
        warnings = []
        errors = []
        
        # 获取排序策略
        if strategy is None:
            strategy_key = self.config.get('sort_strategy', 'priority_first')
            strategy = SortStrategy(strategy_key)
        
        if strategy_params is None and strategy == SortStrategy.WEIGHTED:
            strategy_params = {
                'priority_weight': float(self.config.get('priority_weight', 0.4)),
                'due_weight': float(self.config.get('due_weight', 0.5)),
                'ready_weight': float(self.config.get('ready_weight', 0.1)),
            }
        
        # 创建策略并排序批次
        sorter = StrategyFactory.create(strategy, **(strategy_params or {}))
        batch_list = [
            BatchForSort(
                batch_id=b.batch_id,
                priority=b.priority,
                due_date=b.due_date,
                ready_status=b.ready_status,
                created_at=b.created_at
            )
            for b in batches.values()
        ]
        sorted_batches = sorter.sort(batch_list)
        batch_order = {b.batch_id: i for i, b in enumerate(sorted_batches)}
        
        # 按批次优先级和工序顺序排序工序
        sorted_ops = sorted(
            operations,
            key=lambda op: (batch_order.get(op.batch_id, 999), op.seq)
        )
        
        self.logger.info(
            f"Scheduling {len(sorted_ops)} operations from {len(batches)} batches "
            f"using strategy: {sorter.get_name()}"
        )
        
        # 资源占用追踪
        machine_timeline = {}   # machine_id -> [(start, end), ...]
        operator_timeline = {}  # operator_id -> [(start, end), ...]
        batch_progress = {}     # batch_id -> 最后完成时间
        
        results = []
        scheduled_count = 0
        failed_count = 0
        
        for op in sorted_ops:
            try:
                batch = batches[op.batch_id]
                
                if op.source == 'external':
                    # ─────────────────────────────────────────
                    # 外部供应商：不占资源，只占时间
                    # ─────────────────────────────────────────
                    result = self._schedule_external(op, batch, batch_progress)
                else:
                    # ─────────────────────────────────────────
                    # 内部工序：双重资源约束
                    # ─────────────────────────────────────────
                    result = self._schedule_internal(
                        op, batch, batch_progress,
                        machine_timeline, operator_timeline
                    )
                
                if result:
                    results.append(result)
                    batch_progress[op.batch_id] = result.end_time
                    scheduled_count += 1
                else:
                    failed_count += 1
                    errors.append(f"工序 {op.op_code} 排产失败")
                    
            except Exception as e:
                failed_count += 1
                errors.append(f"工序 {op.op_code} 排产异常: {str(e)}")
                self.logger.error(f"Schedule error for {op.op_code}: {e}")
        
        # 生成摘要
        duration = (datetime.now() - start_time).total_seconds()
        summary = ScheduleSummary(
            success=failed_count == 0,
            total_ops=len(operations),
            scheduled_ops=scheduled_count,
            failed_ops=failed_count,
            warnings=warnings,
            errors=errors,
            duration_seconds=duration
        )
        
        self.logger.info(
            f"Scheduling completed: {scheduled_count}/{len(operations)} ops, "
            f"{failed_count} failed, took {duration:.2f}s"
        )
        
        return results, summary
    
    def _schedule_external(self, op, batch, batch_progress) -> ScheduleResult:
        """排产外部工序"""
        prev_end = batch_progress.get(op.batch_id, datetime.now())
        
        # 计算外部周期
        ext_days = op.ext_days or 1.0
        ext_hours = self._days_to_hours(ext_days)
        
        start = prev_end
        end = self.calendar.add_calendar_days(start, ext_days)
        
        return ScheduleResult(
            op_id=op.id,
            op_code=op.op_code,
            batch_id=op.batch_id,
            seq=op.seq,
            start_time=start,
            end_time=end,
            source='external'
        )
    
    def _schedule_internal(self, op, batch, batch_progress,
                           machine_timeline, operator_timeline) -> ScheduleResult:
        """排产内部工序（双重资源约束）"""
        
        # 1. 前道工序约束
        prev_end = batch_progress.get(op.batch_id, datetime.now())
        
        # 2. 设备可用时间
        machine_available = self._get_resource_available(
            machine_timeline, op.machine_id
        )
        
        # 3. 人员可用时间
        operator_available = self._get_resource_available(
            operator_timeline, op.operator_id
        )
        
        # 4. 取三者最晚时间
        earliest = max(prev_end, machine_available, operator_available)
        
        # 5. 工作日历约束
        earliest = self.calendar.adjust_to_working_time(
            earliest, batch.priority
        )
        
        # 6. 计算工时
        total_hours = op.setup_hours + op.unit_hours * batch.quantity
        
        # 7. 考虑效率系数
        efficiency = self.calendar.get_efficiency(earliest)
        if efficiency and efficiency < 1.0:
            total_hours = total_hours / efficiency
        
        # 8. 计算结束时间
        end = self.calendar.add_working_hours(earliest, total_hours)
        
        # 9. 更新资源占用
        if op.machine_id:
            self._occupy_resource(machine_timeline, op.machine_id, earliest, end)
        if op.operator_id:
            self._occupy_resource(operator_timeline, op.operator_id, earliest, end)
        
        return ScheduleResult(
            op_id=op.id,
            op_code=op.op_code,
            batch_id=op.batch_id,
            seq=op.seq,
            machine_id=op.machine_id,
            operator_id=op.operator_id,
            start_time=earliest,
            end_time=end,
            source='internal'
        )
    
    def _get_resource_available(self, timeline: Dict, resource_id: str) -> datetime:
        """获取资源最早可用时间"""
        if not resource_id:
            return datetime.now()
        if resource_id not in timeline or not timeline[resource_id]:
            return datetime.now()
        
        # 返回最后一个占用时段的结束时间
        return max(end for _, end in timeline[resource_id])
    
    def _occupy_resource(self, timeline: Dict, resource_id: str, 
                         start: datetime, end: datetime):
        """记录资源占用"""
        if not resource_id:
            return
        if resource_id not in timeline:
            timeline[resource_id] = []
        timeline[resource_id].append((start, end))
    
    def _days_to_hours(self, days: float) -> float:
        """天数转换为工作小时"""
        hours_per_day = float(self.config.get('hours_per_day', 8))
        return days * hours_per_day
```

```python
# core/infrastructure/transaction.py

from functools import wraps
from contextlib import contextmanager
import sqlite3
import logging

logger = logging.getLogger(__name__)

class TransactionManager:
    """事务管理器"""
    
    def __init__(self, db_connection):
        self.conn = db_connection
    
    @contextmanager
    def transaction(self):
        """
        事务上下文管理器
        
        使用方式:
            with tx_manager.transaction():
                repo1.create(...)
                repo2.update(...)
                # 自动提交或回滚
        """
        try:
            yield self.conn
            self.conn.commit()
            logger.debug("Transaction committed")
        except Exception as e:
            self.conn.rollback()
            logger.error(f"Transaction rolled back: {e}")
            raise

def transactional(func):
    """
    事务装饰器
    
    使用方式:
        @transactional
        def create_batch_with_operations(self, batch, operations):
            ...
    """
    @wraps(func)
    def wrapper(self, *args, **kwargs):
        with self.tx_manager.transaction():
            return func(self, *args, **kwargs)
    return wrapper


# 使用示例
# ─────────────────────────────────────────────────────────────

class BatchService:
    """批次服务 - 事务示例"""
    
    def __init__(self, batch_repo, op_repo, tx_manager, logger):
        self.batch_repo = batch_repo
        self.op_repo = op_repo
        self.tx_manager = tx_manager
        self.logger = logger
    
    @transactional
    def create_batch_from_template(self, batch_data: dict, part_no: str):
        """
        从模板创建批次（事务保护）
        
        操作:
        1. 创建批次记录
        2. 复制零件工序模板到批次工序
        3. 全部成功才提交，任一失败全部回滚
        """
        # 创建批次
        batch = self.batch_repo.create(batch_data)
        
        # 从模板复制工序
        template_ops = self.op_repo.get_template_by_part(part_no)
        for tmpl in template_ops:
            op_data = {
                'batch_id': batch.batch_id,
                'op_code': f"{batch.batch_id}_{tmpl.seq:02d}",
                'seq': tmpl.seq,
                'op_type_id': tmpl.op_type_id,
                'op_type_name': tmpl.op_type_name,
                'source': tmpl.source,
                'setup_hours': tmpl.setup_hours,
                'unit_hours': tmpl.unit_hours,
                'ext_days': tmpl.ext_days,
            }
            self.op_repo.create(op_data)
        
        self.logger.info(f"Batch {batch.batch_id} created with {len(template_ops)} operations")
        return batch


# 需要事务保护的关键操作清单
# ─────────────────────────────────────────────────────────────

"""
1. 批次创建（批次 + 工序）

# core/infrastructure/locking.py

import sqlite3
import time
import logging
from contextlib import contextmanager
from datetime import datetime, timedelta

logger = logging.getLogger(__name__)

class ResourceLock:
    """资源锁（基于数据库实现）"""
    
    def __init__(self, db_connection):
        self.conn = db_connection
        self._ensure_lock_table()
    
    def _ensure_lock_table(self):
        """确保锁表存在"""
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS ResourceLocks (
                resource_type TEXT NOT NULL,
                resource_id TEXT NOT NULL,
                locked_by TEXT NOT NULL,
                locked_at DATETIME DEFAULT CURRENT_TIMESTAMP,
                expires_at DATETIME NOT NULL,
                PRIMARY KEY (resource_type, resource_id)
            )
        """)
        self.conn.commit()
    
    def acquire(self, resource_type: str, resource_id: str, 
                locked_by: str = 'system', 
                timeout_seconds: int = 300) -> bool:
        """
        获取资源锁
        
        Args:
            resource_type: 资源类型 (如 'schedule', 'batch')
            resource_id: 资源ID
            locked_by: 锁持有者标识
            timeout_seconds: 锁超时时间
        
        Returns:
            是否成功获取锁
        """
        expires_at = datetime.now() + timedelta(seconds=timeout_seconds)
        
        try:
            # 先清理过期锁
            self.conn.execute(
                "DELETE FROM ResourceLocks WHERE expires_at < ?",
                (datetime.now(),)
            )
            
            # 尝试获取锁
            self.conn.execute("""
                INSERT INTO ResourceLocks 
                (resource_type, resource_id, locked_by, expires_at)
                VALUES (?, ?, ?, ?)
            """, (resource_type, resource_id, locked_by, expires_at))
            
            self.conn.commit()
            logger.debug(f"Lock acquired: {resource_type}/{resource_id} by {locked_by}")
            return True
            
        except sqlite3.IntegrityError:
            # 锁已存在
            logger.warning(f"Lock failed: {resource_type}/{resource_id} already locked")
            return False
    
    def release(self, resource_type: str, resource_id: str, 
                locked_by: str = 'system') -> bool:
        """释放资源锁"""
        cursor = self.conn.execute("""
            DELETE FROM ResourceLocks 
            WHERE resource_type = ? AND resource_id = ? AND locked_by = ?
        """, (resource_type, resource_id, locked_by))
        
        self.conn.commit()
        released = cursor.rowcount > 0
        
        if released:
            logger.debug(f"Lock released: {resource_type}/{resource_id}")
        
        return released
    
    @contextmanager
    def lock(self, resource_type: str, resource_id: str,
             locked_by: str = 'system', timeout_seconds: int = 300):
        """
        锁上下文管理器
        
        使用方式:
            with resource_lock.lock('schedule', 'batch_001'):
                # 执行需要锁保护的操作
                pass
        """
        if not self.acquire(resource_type, resource_id, locked_by, timeout_seconds):
            raise ResourceLockError(
                f"无法获取资源锁: {resource_type}/{resource_id}，"
                f"可能有其他操作正在进行"
            )
        
        try:
            yield
        finally:
            self.release(resource_type, resource_id, locked_by)

class ResourceLockError(Exception):
    """资源锁异常"""
    pass

        formatter = logging.Formatter(

        )

        self.logger.addHandler(handler)

```python
# core/infrastructure/backup.py

import os
import shutil
import sqlite3
from datetime import datetime, timedelta
import logging
import schedule
import threading
import time

class BackupManager:
    """数据库备份管理器"""
    
    def __init__(self, 
                 db_path: str,
                 backup_dir: str = "backups",
                 keep_days: int = 7,
                 logger: logging.Logger = None):
        
        self.db_path = db_path
        self.backup_dir = backup_dir
        self.keep_days = keep_days
        self.logger = logger or logging.getLogger(__name__)
        
        # 确保备份目录存在
        os.makedirs(backup_dir, exist_ok=True)
    
    def backup(self, suffix: str = None) -> str:
        """
        执行备份
        
        Args:
            suffix: 备份文件后缀 (可选)
        
        Returns:
            备份文件路径
        """
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        suffix_str = f"_{suffix}" if suffix else ""
        backup_name = f"aps_backup_{timestamp}{suffix_str}.db"
        backup_path = os.path.join(self.backup_dir, backup_name)
        
        try:
            # 使用 SQLite 在线备份 API
            source = sqlite3.connect(self.db_path)
            dest = sqlite3.connect(backup_path)
            
            source.backup(dest)
            
            source.close()
            dest.close()
            
            self.logger.info(f"Database backed up to: {backup_path}")
            return backup_path
            
        except Exception as e:
            self.logger.error(f"Backup failed: {e}")
            raise
    
    def restore(self, backup_path: str) -> bool:
        """
        从备份恢复
        
        Args:
            backup_path: 备份文件路径
        
        Returns:
            是否成功
        """
        if not os.path.exists(backup_path):
            self.logger.error(f"Backup file not found: {backup_path}")
            return False
        
        try:
            # 先备份当前数据库
            self.backup(suffix="before_restore")
            
            # 恢复
            source = sqlite3.connect(backup_path)
            dest = sqlite3.connect(self.db_path)
            
            source.backup(dest)
            
            source.close()
            dest.close()
            
            self.logger.info(f"Database restored from: {backup_path}")
            return True
            
        except Exception as e:
            self.logger.error(f"Restore failed: {e}")
            return False
    
    def cleanup_old_backups(self):
        """清理过期备份"""
        cutoff = datetime.now() - timedelta(days=self.keep_days)
        
        for filename in os.listdir(self.backup_dir):
            if not filename.startswith('aps_backup_'):
                continue
            
            filepath = os.path.join(self.backup_dir, filename)
            file_time = datetime.fromtimestamp(os.path.getmtime(filepath))
            
            if file_time < cutoff:
                os.remove(filepath)
                self.logger.info(f"Deleted old backup: {filename}")
    
    def list_backups(self) -> list:
        """列出所有备份"""
        backups = []
        
        for filename in sorted(os.listdir(self.backup_dir), reverse=True):
            if not filename.startswith('aps_backup_'):
                continue
            
            filepath = os.path.join(self.backup_dir, filename)
            stat = os.stat(filepath)
            
            backups.append({
                'filename': filename,
                'path': filepath,
                'size_mb': round(stat.st_size / 1024 / 1024, 2),
                'created_at': datetime.fromtimestamp(stat.st_mtime).isoformat()
            })
        
        return backups


class BackupScheduler:
    """备份调度器"""
    
    def __init__(self, backup_manager: BackupManager, logger: logging.Logger = None):
        self.backup_manager = backup_manager
        self.logger = logger or logging.getLogger(__name__)
        self._running = False
        self._thread = None
    
    def start(self, 
              daily_time: str = "02:00",
              cleanup_time: str = "03:00"):
        """
        启动定时备份
        
        Args:
            daily_time: 每日备份时间 (HH:MM)
            cleanup_time: 每日清理时间 (HH:MM)
        """
        # 每日备份
        schedule.every().day.at(daily_time).do(self._daily_backup)
        
        # 每日清理
        schedule.every().day.at(cleanup_time).do(self.backup_manager.cleanup_old_backups)
        
        # 后台线程运行
        self._running = True
        self._thread = threading.Thread(target=self._run_scheduler, daemon=True)
        self._thread.start()
        
        self.logger.info(f"Backup scheduler started. Daily backup at {daily_time}")
    
    def stop(self):
        """停止调度"""
        self._running = False
        if self._thread:
            self._thread.join(timeout=5)
        self.logger.info("Backup scheduler stopped")
    
    def _run_scheduler(self):
        """调度器主循环"""
        while self._running:
            schedule.run_pending()
            time.sleep(60)
    
    def _daily_backup(self):
        """执行每日备份"""
        try:
            self.backup_manager.backup(suffix="daily")
        except Exception as e:
            self.logger.error(f"Daily backup failed: {e}")

# ─────────────────────────────────────────────────────────────

```python
# core/services/common/excel_service.py

import pandas as pd
from typing import List, Dict, Tuple, Any, Callable
from dataclasses import dataclass, field
from enum import Enum
import logging

class ImportMode(Enum):
    APPEND = "append"       # 追加，不影响已有
    OVERWRITE = "overwrite" # 相同ID覆盖
    REPLACE = "replace"     # 清空后导入

class RowStatus(Enum):
    NEW = "new"
    UPDATE = "update"
    UNCHANGED = "unchanged"
    ERROR = "error"
    SKIP = "skip"

@dataclass
class ImportPreviewRow:
    """导入预览行"""
    row_num: int
    status: RowStatus
    data: Dict[str, Any]
    message: str = ""
    changes: Dict[str, Tuple] = field(default_factory=dict)  # {field: (old, new)}

@dataclass
class ImportResult:
    """导入结果"""
    success: bool
    total: int
    new_count: int
    update_count: int
    skip_count: int
    error_count: int
    errors: List[Dict]  # [{row: int, field: str, message: str}]
    warnings: List[str]

class ExcelService:
    """通用Excel导入导出服务"""
    
    def __init__(self, template_dir: str, logger: logging.Logger = None):
        self.template_dir = template_dir
        self.logger = logger or logging.getLogger(__name__)
    
    def read_excel(self, file_path: str, sheet_name: str = None) -> pd.DataFrame:
        """读取Excel文件"""
        try:
            df = pd.read_excel(file_path, sheet_name=sheet_name or 0)
            self.logger.info(f"Read Excel: {file_path}, {len(df)} rows")
            return df
        except Exception as e:
            self.logger.error(f"Failed to read Excel: {e}")
            raise
    
    def write_excel(self, data: List[Dict], file_path: str, 
                    columns: List[str] = None,
                    column_names: Dict[str, str] = None):
        """
        写入Excel文件
        
        Args:
            data: 数据列表
            file_path: 输出路径
            columns: 列顺序
            column_names: 列名映射 {英文: 中文}
        """
        try:
            df = pd.DataFrame(data)
            if columns:
                df = df[columns]
            if column_names:
                df = df.rename(columns=column_names)
            df.to_excel(file_path, index=False)
            self.logger.info(f"Write Excel: {file_path}, {len(df)} rows")
        except Exception as e:
            self.logger.error(f"Failed to write Excel: {e}")
            raise
    
    def preview_import(self, 
                       df: pd.DataFrame,
                       id_column: str,
                       existing_data: Dict[str, Any],
                       validators: List[Callable] = None,
                       mode: ImportMode = ImportMode.OVERWRITE) -> List[ImportPreviewRow]:
        """
        生成导入预览
        
        Args:
            df: 待导入数据
            id_column: ID列名
            existing_data: 已存在数据 {id: record}
            validators: 验证函数列表
            mode: 导入模式
        
        Returns:
            预览行列表
        """
        preview = []
        
        for idx, row in df.iterrows():
            row_num = idx + 2  # Excel行号 (1-based + 标题行)
            row_dict = row.to_dict()
            row_id = row_dict.get(id_column)
            
            # 空ID检查
            if pd.isna(row_id) or str(row_id).strip() == '':
                preview.append(ImportPreviewRow(
                    row_num=row_num,
                    status=RowStatus.ERROR,
                    data=row_dict,
                    message=f'{id_column}不能为空'
                ))
                continue
            
            row_id = str(row_id).strip()
            row_dict[id_column] = row_id
            
            # 执行验证
            validation_error = None
            if validators:
                for validator in validators:
                    error = validator(row_dict)
                    if error:
                        validation_error = error
                        break
            
            if validation_error:
                preview.append(ImportPreviewRow(
                    row_num=row_num,
                    status=RowStatus.ERROR,
                    data=row_dict,
                    message=validation_error
                ))
                continue
            
            # 判断状态
            if row_id in existing_data:
                if mode == ImportMode.APPEND:
                    preview.append(ImportPreviewRow(
                        row_num=row_num,
                        status=RowStatus.SKIP,
                        data=row_dict,
                        message='已存在，跳过'
                    ))
                else:
                    # 计算变更
                    existing = existing_data[row_id]
                    changes = self._calc_changes(existing, row_dict)
                    
                    if changes:
                        preview.append(ImportPreviewRow(
                            row_num=row_num,
                            status=RowStatus.UPDATE,
                            data=row_dict,
                            message='将更新现有记录',
                            changes=changes
                        ))
                    else:
                        preview.append(ImportPreviewRow(
                            row_num=row_num,
                            status=RowStatus.UNCHANGED,
                            data=row_dict,
                            message='无变更'
                        ))
            else:
                preview.append(ImportPreviewRow(
                    row_num=row_num,
                    status=RowStatus.NEW,
                    data=row_dict,
                    message='新增记录'
                ))
        
        return preview
    
    def _calc_changes(self, existing: Any, new_data: Dict) -> Dict[str, Tuple]:
        """计算变更"""
        changes = {}
        existing_dict = existing if isinstance(existing, dict) else vars(existing)
        
        for key, new_val in new_data.items():
            if key in existing_dict:
                old_val = existing_dict[key]
                # 处理空值比较
                if pd.isna(new_val):
                    new_val = None
                if old_val != new_val:
                    changes[key] = (old_val, new_val)
        
        return changes
    
    def get_template_path(self, template_name: str) -> str:
        """获取模板文件路径"""
        return os.path.join(self.template_dir, f"{template_name}.xlsx")
    
    def generate_template(self, 
                          columns: List[Dict[str, str]],
                          file_path: str,
                          sample_data: List[Dict] = None):
        """
        生成导入模板
        
        Args:
            columns: [{name: str, title: str, required: bool, example: str}]
            file_path: 输出路径
            sample_data: 示例数据
        """
        # 创建标题行
        headers = [col['title'] for col in columns]
        
        df = pd.DataFrame(columns=headers)
        
        # 添加示例数据
        if sample_data:
            for data in sample_data:
                row = [data.get(col['name'], '') for col in columns]
                df.loc[len(df)] = row
        
        df.to_excel(file_path, index=False)
        self.logger.info(f"Template generated: {file_path}")
```

│     → 外部工序组：连续外部合并周期                          │
│     → 删除规则：首尾可删、中间不可删                        │
│     → 双重资源约束：设备+人员                               │

关键事务边界：
─────────────────────────────────────────────────────────────
  1. 批次创建 = 批次记录 + 工序记录 (原子)
  2. 批次删除 = 排程记录 + 工序记录 + 批次记录 (原子)
  3. 排产执行 = 多条排程记录 (原子)
  4. 工艺模板保存 = 零件 + 工序 + 外部组 (原子)
  5. Excel批量导入 = 多条记录 (原子)
